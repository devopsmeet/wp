pipeline {
  environment {
    registry = "devopsmeet2019/wp-project"
    registryCredential = 'DockerLogin'
    dockerImage = ''
  }
  agent any
  stages {
    stage('Cloning Git') {
      steps {
        git 'https://github.com/devopsmeet/wp.git'
      }
    }
    stage('Building image') {
      steps{
        script {
          dockerImage = docker.build registry + ":$BUILD_NUMBER"
        }
      }
    }
    stage('Deploy Image') {
      steps{
        script {
          docker.withRegistry( '', registryCredential ) {
            dockerImage.push()
          }
        }
      }
    }
    stage('Deploy latest image on container') {
      steps{
        script {
        docker.image('$registry:$BUILD_NUMBER').withRun('-p 9090:80') {c ->
          }
       }
    }    
  }   
  } 
}
